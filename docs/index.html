insim.js

InSim.js

A TypeScript version of the InSim.NET library

Installing

 
$ yarn add insim.js

Example

 
import { InSim, PacketType } from 'insim.js';

const insim = new InSim({
  connection: {
    host: '127.0.0.1',
    port: 29999,
  },

  insimOptions: {
    name: 'InSim.js',
    prefix: '!',
  },
});

insim.on('error', error => {
  console.log('Error', error);
});

insim.bind<IS_MSO>(PacketType.ISP_MSO, (data) => {
  console.log('Chat message', data);
});

insim.connect();
